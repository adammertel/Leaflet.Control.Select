# Grouping options

A basic example using groups of options to create a hierarchical menu

**list of items:**

```js
var places = [
  {
    label: "towns",
    value: "town group",
    items: [
      { label: "Nov치 Dubnica", value: [[48.936649, 18.141319], 16] },
      { label: "Dubnica nad V치hom", value: [[48.959672, 18.170311], 16] },
      { label: "Nem코ov치", value: [[48.965817, 18.121432], 16] }
    ]
  },
  {
    label: "villages",
    value: "villages group",
    items: [
      {
        label: "north",
        value: "north-villages-group",
        items: [
          { label: "Bole코ov", value: [[48.987006, 18.15625], 16] },
          { label: "Bor캜ice", value: [[48.979288, 18.137024], 16] },
          { label: "Kameni캜any", value: [[48.991538, 18.170283], 16] }
        ]
      },
      {
        label: "west",
        value: "west-villages-group",
        items: [
          { label: "Doln치 S칰캜a", value: [[48.957633, 18.031684], 16] },
          {
            label: "Skalka nad V치hom",
            value: [[48.92652, 18.071181], 16]
          }
        ]
      },
      {
        label: "south",
        value: "south-villages-group",
        items: [
          { label: "Kola캜칤n", value: [[48.936836, 18.16759], 16] },
          {
            label: "Tren캜ianska Tepl치",
            value: [[48.93461, 18.120773], 16]
          }
        ]
      }
    ]
  },
  {
    label: "region",
    value: [[48.945548, 18.12225], 13]
  }
];
```

**initialization of the L.Control.Select:**

```js
L.control
  .select({
    position: "topleft",
    id: "image-selector",
    selectedDefault: false,
    items: places,
    iconMain: "游끶",
    onSelect: function(newPosition) {
      drawMarker(newPosition[0], newPosition[1]);
    },
    onGroupOpen: function(groupOpened) {
      // console.log(groupOpened)
    }
  })
  .addTo(map);
```

**onSelect function:**

```js
var recenter = function(newCenter, newZoom) {
  map.setView(newCenter, newZoom);
};
```
